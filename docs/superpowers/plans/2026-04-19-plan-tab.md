# Plan Tab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a YNAB-style `/plan` tab that lets the user assign a max per category per month, showing live `Assigned − Spent = Available` and a "Ready to Assign" banner driven by monthly income. No rollover. Ships on both desktop (SQLite) and web (PostgreSQL).

**Architecture:** Single new `monthly_budgets` table (mirrored in `db-sqlite` and `db-pg`). New `plan.ts` query module (`getMonthPlan`) and action module (`setCategoryAssigned`, `clearCategoryAssigned`) per DB package. Shared React UI in `@florin/core/components/plan` consumed by both `apps/web` and `apps/desktop` routes.

**Tech Stack:** TypeScript, Drizzle ORM (SQLite via better-sqlite3, PostgreSQL via postgres.js), Next.js 15 App Router, React 19, Tailwind, Vitest (for DB query tests), Zod (action input validation), lucide-react (icons).

**Spec:** [`docs/superpowers/specs/2026-04-19-plan-tab-design.md`](../specs/2026-04-19-plan-tab-design.md)

**Pragmatic testing note:** The repo has Vitest set up in `apps/web/tests/unit/` (node env, no jsdom / RTL). This plan uses Vitest + in-memory SQLite for DB-layer TDD. React component tests require RTL/jsdom setup which is out of scope — UI is verified through type-checking, manual smoke, and a Playwright E2E that's added if Playwright ever lands in the repo (currently not configured — task 11 flags this).

```ts
export interface PlanCategory {
  /** categories.id */
  id: string
  name: string
  emoji: string | null
  /** monthly_budgets.assigned for this (year, month, category). 0 if no row. */
  assigned: number
  /** Sum of ABS(amount) for non-transfer, non-deleted transactions this month. */
  spent: number
  /** assigned - spent. Negative = overspent. */
  available: number
  /** monthly_budgets.note. null if no row. */
  note: string | null
}

export interface PlanGroup {
  /** category_groups.id */
  id: string
  name: string
  kind: 'income' | 'expense'
  color: string | null
  categories: PlanCategory[]
  /** Sum of child .assigned. */
  assigned: number
  /** Sum of child .spent. */
  spent: number
  /** Sum of child .available. */
  available: number
  /** Count of child categories with available < 0. */
  overspentCount: number
}

export interface MonthPlan {
  year: number
  month: number
  /** Only expense groups appear here — income groups feed `income` below. */
  groups: PlanGroup[]
  /** Sum of all transactions in income-kind categories this month, transfers/soft-deletes excluded. */
  income: number
  /** Sum of .assigned across every PlanCategory. */
  totalAssigned: number
  /** income - totalAssigned. Negative = "Assigned Too Much". */
  readyToAssign: number
  /** Total count of overspent categories across all expense groups. */
  overspentCount: number
}

export interface SetCategoryAssignedInput {
  year: number
  month: number
  categoryId: string
  amount: number
  note?: string | null
}
```

```ts
import { and, asc, eq, gte, isNull, lt, sql } from 'drizzle-orm'
import type { MonthPlan, PlanCategory, PlanGroup } from '@florin/core/types'
import type { SqliteDB } from '../client'
import { categories, categoryGroups, monthlyBudgets, transactions } from '../schema'

/**
 * Compute the full Plan view for a given (year, month).
 *
 * Three raw sub-queries, stitched together in memory:
 *   1. All non-archived categories + their group + the assigned amount (LEFT JOIN budgets).
 *   2. Sum of spent per category for the month (expense groups only, no transfers, no deletes).
 *   3. Sum of income for the month (income groups, same filters).
 *
 * We intentionally do this synchronously with better-sqlite3 — the query
 * layer returns a plain Promise to stay compatible with the async
 * FlorinQueries interface.
 */
export function getMonthPlanQuery(
  db: SqliteDB,
  year: number,
  month: number,
): MonthPlan {
  const startDate = monthStartIso(year, month)
  const endDate = monthStartIso(year, month + 1) // handles month=12 → year+1 jan

  // 1. Categories + groups + assigned.
  const rows = db
    .select({
      catId: categories.id,
      catName: categories.name,
      catEmoji: categories.emoji,
      catArchived: categories.isArchived,
      catOrder: categories.displayOrder,
      groupId: categoryGroups.id,
      groupName: categoryGroups.name,
      groupKind: categoryGroups.kind,
      groupColor: categoryGroups.color,
      groupOrder: categoryGroups.displayOrder,
      assigned: monthlyBudgets.assigned,
      note: monthlyBudgets.note,
    })
    .from(categories)
    .innerJoin(categoryGroups, eq(categoryGroups.id, categories.groupId))
    .leftJoin(
      monthlyBudgets,
      and(
        eq(monthlyBudgets.categoryId, categories.id),
        eq(monthlyBudgets.year, year),
        eq(monthlyBudgets.month, month),
      ),
    )
    .where(eq(categories.isArchived, false))
    .orderBy(asc(categoryGroups.displayOrder), asc(categoryGroups.name), asc(categories.displayOrder), asc(categories.name))
    .all()

  // 2. Spent per category for the month (expense categories only).
  const spentRows = db
    .select({
      categoryId: transactions.categoryId,
      total: sql<number>`COALESCE(SUM(ABS(${transactions.amount})), 0)`,
    })
    .from(transactions)
    .innerJoin(categories, eq(categories.id, transactions.categoryId))
    .innerJoin(categoryGroups, eq(categoryGroups.id, categories.groupId))
    .where(
      and(
        gte(transactions.occurredAt, startDate),
        lt(transactions.occurredAt, endDate),
        isNull(transactions.deletedAt),
        sql`${transactions.transferPairId} IS NULL`,
        eq(categoryGroups.kind, 'expense'),
      ),
    )
    .groupBy(transactions.categoryId)
    .all()

  const spentByCategory = new Map<string, number>()
  for (const r of spentRows) {
    if (r.categoryId) spentByCategory.set(r.categoryId, Number(r.total))
  }

  // 3. Income sum for the month.
  const incomeRow = db
    .select({
      total: sql<number>`COALESCE(SUM(${transactions.amount}), 0)`,
    })
    .from(transactions)
    .innerJoin(categories, eq(categories.id, transactions.categoryId))
    .innerJoin(categoryGroups, eq(categoryGroups.id, categories.groupId))
    .where(
      and(
        gte(transactions.occurredAt, startDate),
        lt(transactions.occurredAt, endDate),
        isNull(transactions.deletedAt),
        sql`${transactions.transferPairId} IS NULL`,
        eq(categoryGroups.kind, 'income'),
      ),
    )
    .get()

  const income = Number(incomeRow?.total ?? 0)

  // Stitch — group by expense groups only.
  const groupsById = new Map<string, PlanGroup>()

  for (const r of rows) {
    if (r.groupKind !== 'expense') continue

    let g = groupsById.get(r.groupId)
    if (!g) {
      g = {
        id: r.groupId,
        name: r.groupName,
        kind: 'expense',
        color: r.groupColor,
        categories: [],
        assigned: 0,
        spent: 0,
        available: 0,
        overspentCount: 0,
      }
      groupsById.set(r.groupId, g)
    }

    const assigned = Number(r.assigned ?? 0)
    const spent = spentByCategory.get(r.catId) ?? 0
    const available = assigned - spent

    const cat: PlanCategory = {
      id: r.catId,
      name: r.catName,
      emoji: r.catEmoji,
      assigned,
      spent,
      available,
      note: r.note,
    }

    g.categories.push(cat)
    g.assigned += assigned
    g.spent += spent
    g.available += available
    if (available < 0) g.overspentCount += 1
  }

  // Preserve the SQL ordering we asked for (display_order asc).
  // Iterating the rows in order already gives us groups in the right order
  // because the first time each groupId appears defines insertion order.
  const groups = Array.from(groupsById.values())

  const totalAssigned = groups.reduce((s, g) => s + g.assigned, 0)
  const readyToAssign = income - totalAssigned
  const overspentCount = groups.reduce((s, g) => s + g.overspentCount, 0)

  return {
    year,
    month,
    groups,
    income,
    totalAssigned,
    readyToAssign,
    overspentCount,
  }
}

function monthStartIso(year: number, month: number): string {
  // month 1-12 inclusive. Returns 'YYYY-MM-01'. Handles month=13 as next year's Jan.
  let y = year
  let m = month
  while (m > 12) {
    m -= 12
    y += 1
  }
  while (m < 1) {
    m += 12
    y -= 1
  }
  const mm = String(m).padStart(2, '0')
  return `${y}-${mm}-01`
}
```

```ts
import { and, eq, sql } from 'drizzle-orm'
import { z } from 'zod'
import type { ActionResult, SetCategoryAssignedInput } from '@florin/core/types'
import type { SqliteDB } from '../client'
import { monthlyBudgets } from '../schema'

const setCategoryAssignedSchema = z.object({
  year: z.number().int().min(1970).max(9999),
  month: z.number().int().min(1).max(12),
  categoryId: z.string().min(1),
  amount: z.number().finite().min(0, 'amount must be greater than or equal to 0'),
  note: z.string().max(500).optional().nullable(),
})

export async function setCategoryAssignedMutation(
  db: SqliteDB,
  input: SetCategoryAssignedInput,
): Promise<ActionResult> {
  const parsed = setCategoryAssignedSchema.safeParse(input)
  if (!parsed.success) {
    return {
      success: false,
      error: parsed.error.issues.map((i) => i.message).join(', '),
    }
  }
  const { year, month, categoryId, amount, note } = parsed.data

  try {
    await db
      .insert(monthlyBudgets)
      .values({
        year,
        month,
        categoryId,
        assigned: amount,
        note: note ?? null,
      })
      .onConflictDoUpdate({
        target: [monthlyBudgets.year, monthlyBudgets.month, monthlyBudgets.categoryId],
        set: {
          assigned: amount,
          note: note ?? null,
          updatedAt: sql`(datetime('now'))`,
        },
      })
    return { success: true }
  } catch (error: unknown) {
    return {
      success: false,
      error: error instanceof Error ? error.message : 'Failed to set assigned amount',
    }
  }
}

export async function clearCategoryAssignedMutation(
  db: SqliteDB,
  year: number,
  month: number,
  categoryId: string,
): Promise<ActionResult> {
  try {
    await db
      .delete(monthlyBudgets)
      .where(
        and(
          eq(monthlyBudgets.year, year),
          eq(monthlyBudgets.month, month),
          eq(monthlyBudgets.categoryId, categoryId),
        ),
      )
    return { success: true }
  } catch (error: unknown) {
```

- [ ] **Step 1: Write the PG version**

Create `packages/db-pg/src/actions/plan.ts`. Same Zod schema. For the upsert: drizzle-orm's PG driver uses `.onConflictDoUpdate` with `target` in the same way as SQLite. For the timestamp update, use `sql\`now()\`` instead of `datetime('now')`. Numeric values: `assigned` is a `numeric` column in PG, so pass it as a string: `String(amount)`.

```ts
import { and, eq, sql } from 'drizzle-orm'
import { z } from 'zod'
import type { ActionResult, SetCategoryAssignedInput } from '@florin/core/types'
import type { PgDB } from '../client'
import { monthlyBudgets } from '../schema'

const setCategoryAssignedSchema = z.object({
  year: z.number().int().min(1970).max(9999),
  month: z.number().int().min(1).max(12),
  categoryId: z.string().min(1),
  amount: z.number().finite().min(0, 'amount must be greater than or equal to 0'),
  note: z.string().max(500).optional().nullable(),
})

export async function setCategoryAssignedMutation(
  db: PgDB,
  input: SetCategoryAssignedInput,
): Promise<ActionResult> {
  const parsed = setCategoryAssignedSchema.safeParse(input)
  if (!parsed.success) {
    return {
      success: false,
      error: parsed.error.issues.map((i) => i.message).join(', '),
    }
  }
  const { year, month, categoryId, amount, note } = parsed.data

  try {
    await db
      .insert(monthlyBudgets)
      .values({
        year,
        month,
        categoryId,
        assigned: String(amount),
        note: note ?? null,
      })
      .onConflictDoUpdate({
        target: [monthlyBudgets.year, monthlyBudgets.month, monthlyBudgets.categoryId],
        set: {
          assigned: String(amount),
          note: note ?? null,
          updatedAt: sql`now()`,
        },
      })
    return { success: true }
  } catch (error: unknown) {
    return {
      success: false,
      error: error instanceof Error ? error.message : 'Failed to set assigned amount',
    }
  }
}

export async function clearCategoryAssignedMutation(
  db: PgDB,
  year: number,
  month: number,
  categoryId: string,
): Promise<ActionResult> {
  try {
    await db
      .delete(monthlyBudgets)
      .where(
        and(
          eq(monthlyBudgets.year, year),
          eq(monthlyBudgets.month, month),
          eq(monthlyBudgets.categoryId, categoryId),
        ),
      )
    return { success: true }
  } catch (error: unknown) {
    return {
      success: false,
      error: error instanceof Error ? error.message : 'Failed to clear assigned amount',
    }
  }
}
```

```tsx
  function shift(delta: number) {
```

```tsx
'use client'

import { useState, useTransition } from 'react'
import { useRouter, useSearchParams } from 'next/navigation'
import { toast } from 'sonner'
import type { ActionResult, MonthPlan, SetCategoryAssignedInput } from '@florin/core/types'
import { MonthPicker } from './MonthPicker'
import { PlanBanner } from './PlanBanner'
import { PlanGroup } from './PlanGroup'

interface PlanPageProps {
  plan: MonthPlan
  currency: string
  onSetAssigned: (input: SetCategoryAssignedInput) => Promise<ActionResult>
}

export function PlanPage({ plan, currency, onSetAssigned }: PlanPageProps) {
  const router = useRouter()
  const searchParams = useSearchParams()
  const [, startTransition] = useTransition()
  const [optimistic, setOptimistic] = useState<MonthPlan>(plan)

  // If the server plan changes (e.g. after navigating months), reset optimistic state.
  if (optimistic !== plan && optimistic.year === plan.year && optimistic.month === plan.month) {
    // same month, server refreshed — ignore (our optimistic wins until next nav)
  }
  if (optimistic.year !== plan.year || optimistic.month !== plan.month) {
    setOptimistic(plan)
  }

  function navigate(year: number, month: number) {
    const params = new URLSearchParams(searchParams?.toString() ?? '')
    params.set('month', `${year}-${String(month).padStart(2, '0')}`)
    router.push(`?${params.toString()}`)
  }

  async function handleAssignedChange(categoryId: string, amount: number) {
    // Optimistic update: recompute the in-memory plan with the new assigned.
    setOptimistic((prev) => recomputeWithAssigned(prev, categoryId, amount))

    const result = await onSetAssigned({
      year: plan.year,
      month: plan.month,
      categoryId,
      amount,
    })

    if (!result.success) {
      toast.error(result.error ?? 'Failed to save assignment')
      // Revert: ask Next.js to re-fetch the server plan.
      startTransition(() => router.refresh())
      return
    }
    startTransition(() => router.refresh())
  }

  return (
    <div className="flex flex-col h-full min-h-0">
      <MonthPicker year={optimistic.year} month={optimistic.month} onChange={navigate} />
      <PlanBanner plan={optimistic} currency={currency} />
      <div className="flex-1 overflow-y-auto">
        {optimistic.groups.map((g) => (
          <PlanGroup
            key={g.id}
            group={g}
            currency={currency}
            onAssignedChange={handleAssignedChange}
          />
        ))}
      </div>
    </div>
  )
}

function recomputeWithAssigned(plan: MonthPlan, categoryId: string, newAmount: number): MonthPlan {
  let totalAssigned = 0
  let overspentCount = 0
  const groups = plan.groups.map((g) => {
    let gAssigned = 0
    let gAvailable = 0
    let gOverspent = 0
    const categories = g.categories.map((c) => {
      const assigned = c.id === categoryId ? newAmount : c.assigned
      const available = assigned - c.spent
      gAssigned += assigned
      gAvailable += available
      if (available < 0) gOverspent += 1
      return { ...c, assigned, available }
    })
    totalAssigned += gAssigned
    overspentCount += gOverspent
    return { ...g, categories, assigned: gAssigned, available: gAvailable, overspentCount: gOverspent }
  })
  return {
    ...plan,
    groups,
    totalAssigned,
    readyToAssign: plan.income - totalAssigned,
    overspentCount,
  }
}
```
